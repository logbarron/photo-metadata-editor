# Photo Metadata Editor Configuration
# Copy this file to .env and keep it in the /code/ directory alongside
# photo_metadata_editor.py

# ============================================================================
# CAMERA / SCANNER INFORMATION
# ============================================================================
# These values are written to photos that do not already have metadata.
# Set them to the device that digitized the photo.
CAMERA_MAKE=FUJIFILM
CAMERA_MODEL=GFX100S II
IMAGE_DESCRIPTION=Camera Scanned Image

# Example for an Epson scanner:
# CAMERA_MAKE=Epson
# CAMERA_MODEL=Perfection V850 Pro
# IMAGE_DESCRIPTION=Epson Scan

# ============================================================================
# METADATA DEFAULTS
# ============================================================================
# These keywords are added to photos that are still missing metadata.
# They appear in Apple Photos and make incomplete photos easy to find later.
DATE_KEYWORD=MissingDate
LOCATION_KEYWORD=MissingLocation

# ============================================================================
# UNKNOWN DATE HANDLING
# ============================================================================
# Default date used when you click the "Unknown Date" button.
# 1901-01-02 is useful because Apple Photos groups unknowns together.
UNKNOWN_YEAR=1901
UNKNOWN_MONTH=01
UNKNOWN_DAY=02

# ============================================================================
# PERFORMANCE SETTINGS
# ============================================================================
# Parallel workers for thumbnail generation and metadata processing.
# Lower these if your machine is short on CPU or memory.
THUMBNAIL_WORKERS=10
METADATA_WORKERS=10

# In-memory cache sizes for metadata and thumbnails.
# Increase these for larger libraries if you have enough RAM.
METADATA_CACHE_SIZE=1000
THUMBNAIL_CACHE_SIZE=1000

# ============================================================================
# APPLICATION SETTINGS
# ============================================================================
# Web interface port.
WEB_PORT=5555

# ============================================================================
# EXTERNAL TOOLS
# ============================================================================
# ExifTool version to download automatically if it is not already present.
EXIFTOOL_VERSION=13.30

# ============================================================================
# LLM PARSER (OPTIONAL)
# ============================================================================
# Enables LLM-based filename parsing for date and location suggestions.
# Set to false to skip the model download and use basic pattern matching instead.
LLM_PARSER_ENABLED=true
